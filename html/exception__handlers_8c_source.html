139 void print_data_abort()
140 {
141  u32 dfsr = read_dfsr();
142  u32 dfar = read_dfar();
143  u32 fault_status = (dfsr & 0xF) | ((dfsr & 0x400) >> 6);
144 
145  sw_printk("SW: Data Abort Address: %08x\n", dfar);
146  sw_printk("SW: Fault type: ");
147  sw_printk((char*)data_abt_fault_string[fault_status]);
148  sw_printk("SW: (%x), domain %x, Write not Read: %x, External: %x\n",
149  fault_status, ((dfsr & FSR_DOMAIN_MASK) >> 4), ((dfsr & FSR_WNR_MASK) >> 11), ((dfsr & FSR_EXT_MASK) >> 12 ));
150 }

258  print_data_abort();

260  switch(fault_status) {

276  case dfs_permission_section:
277  case dfs_translation_table_walk_lvl2_sync_ext_abt:
278  case dfs_permission_page:
