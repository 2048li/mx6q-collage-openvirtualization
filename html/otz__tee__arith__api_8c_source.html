490 void TEE_BigIntDivide(TEE_BigInt* dest_q, TEE_BigInt* dest_r, TEE_BigInt* op1, TEE_BigInt* op2){
491  TEE_BigInt *tmp1, *shifted, *adder;
492  int diffBit;
493  tmp1 = (TEE_BigInt*)TEE_Malloc(TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) * sizeof(TEE_BigInt), 0);
494  shifted = (TEE_BigInt*)TEE_Malloc(TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) * sizeof(TEE_BigInt), 0);
495  TEE_MemMove(tmp1,op1,(TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) * sizeof(TEE_BigInt)));
496  while(TEE_BigIntCmp(tmp1,op2) >= 0){
497  diffBit = MaxBitPos(tmp1) - MaxBitPos(op2);
498  TEE_MemMove(shifted,op2,(TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) * sizeof(TEE_BigInt)));
499  if(diffBit){
500  TEE_BigIntLeftShift(shifted, shifted, diffBit);
501  adder = (TEE_BigInt*)TEE_Malloc(TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) * sizeof(TEE_BigInt), 0);
502  if(TEE_BigIntCmp(tmp1,shifted) >=0){
503  *(adder+TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) - 1 - (diffBit/32)) = XPowerY(2, diffBit%32);
504  TEE_BigIntAdd(dest_q, dest_q, adder);
505  TEE_BigIntSub(tmp1, tmp1, shifted);
506  }else if(diffBit - 1){
507  TEE_BigIntLeftShift(shifted, shifted, diffBit - 1);
508  *(adder+TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) - 1 - (diffBit/31)) = XPowerY(2, (diffBit%31) - 1);
509  TEE_BigIntAdd(dest_q, dest_q, adder);
510  TEE_BigIntSub(tmp1, tmp1, shifted);
511  }else{
512  TEE_BigIntSub(tmp1, tmp1, op2);
513  incrementBigInt(dest_q);
514  }
515  }else{
516  TEE_BigIntSub(tmp1, tmp1, op2);
517  incrementBigInt(dest_q);
518  }
519  }
520  TEE_MemMove(dest_r,tmp1,(TEE_BigIntSizeInU32(TEE_BigIntNumberOfBits) * sizeof(TEE_BigInt)));
521 }
