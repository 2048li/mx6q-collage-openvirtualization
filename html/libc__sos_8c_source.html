101 void* realloc(void* ptr, u32 size)
102 {
103  void* local_ptr = NULL;
104  u32 old_size = 0;
105 
106  if(ptr == NULL) {
107  return(sw_malloc(size));
108  }
109  if((size == 0) && (ptr != NULL)) {
110  sw_free(ptr);
111  return(NULL);
112  }
113  local_ptr = sw_malloc(size);
114  old_size = get_ptr_size(ptr);
115  sw_memcpy(local_ptr,ptr,MIN(old_size,size));
116  sw_free(ptr);
117  return(local_ptr);
118 }
